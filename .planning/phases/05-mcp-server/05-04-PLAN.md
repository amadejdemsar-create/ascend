```typescript
function formatTree(goals: any[], indent: number = 0): string {
  return goals.map(g => {
    const prefix = "  ".repeat(indent);
    const status = g.status === "COMPLETED" ? "x" : " ";
    const line = `${prefix}- [${status}] ${g.title} (${g.horizon}, ${g.priority}, ${g.progress}%)`;
    const children = g.children?.length ? "\n" + formatTree(g.children, indent + 1) : "";
    return line + children;
  }).join("\n");
}
```

Return the formatted tree followed by raw JSON.
