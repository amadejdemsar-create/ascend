  <DragDropProvider
    onDragStart={() => { snapshot.current = { ...items }; }}
    onDragOver={(event) => {
      const { source } = event.operation;
      if (source?.type === 'column') return;
      setItems((prev) => move(prev, event));
    }}
    onDragEnd={(event) => {
      if (event.canceled) {
        setItems(snapshot.current);
        return;
      }
      // Detect what changed
      const source = event.operation?.source;
      if (!source) return;
      const goalId = String(source.id);
      // Find new column
      let newCol: string | null = null;
      for (const [col, colGoals] of Object.entries(items)) {
        if (colGoals.some(g => g.id === goalId)) { newCol = col; break; }
      }
      let oldCol: string | null = null;
      for (const [col, colGoals] of Object.entries(snapshot.current)) {
        if (colGoals.some(g => g.id === goalId)) { oldCol = col; break; }
      }
      // Cross-column: update field
      if (newCol && oldCol && newCol !== oldCol) {
        const field = boardGroupBy === "status" ? "status"
          : boardGroupBy === "horizon" ? "horizon"
          : "categoryId";
        const value = boardGroupBy === "category" && newCol === "uncategorized" ? null : newCol;
        const data: Record<string, unknown> = { [field]: value };

        );
      }
      // Persist sortOrder within the target column
      if (newCol && items[newCol]) {
        reorderMutation.mutate(items[newCol].map((g, i) => ({ id: g.id, sortOrder: i })));
      }
    }}
  >
    {/* Grouping toggle + column grid */}

      {(source) => {
        const goal = goals.find(g => g.id === String(source?.id));
        if (!goal) return null;
        return <GoalDragOverlay goal={{ id: goal.id, title: goal.title, priority: goal.priority, category: goal.category ? { name: goal.category.name, color: goal.category.color } : null }} />;

  </DragDropProvider>
